 contract Map {
     address public owner;
     uint256[] map;

     function set(uint256 key, uint256 value) public {
         if (map.length <= key) {
             map.length = key + 1;
         }
        
         map[key] = value;
     }

     function get(uint256 key) public view returns (uint256) {

     }
